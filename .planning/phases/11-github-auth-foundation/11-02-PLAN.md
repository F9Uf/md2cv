    ```typescript
    export interface GitHubUser {
      login: string
      avatar_url: string
    }

    export interface CallbackParams {
      code: string | null
      state: string | null
      error: string | null
    }

    const OAUTH_SCOPE = 'repo' // D-11: read/write private + public repos for Phase 12 commits

    export function buildAuthorizeUrl(clientId: string, state: string, redirectUri: string): string {
      const params = new URLSearchParams({
        client_id: clientId,
        redirect_uri: redirectUri,
        scope: OAUTH_SCOPE,
        state,
      })
      return `https://github.com/login/oauth/authorize?${params.toString()}`
    }

    export function generateState(): string {
      const bytes = new Uint8Array(16)
      crypto.getRandomValues(bytes)
      return Array.from(bytes, (b) => b.toString(16).padStart(2, '0')).join('')
    }

    export function parseCallbackParams(search: string): CallbackParams {
      const params = new URLSearchParams(search)
      return {
        code: params.get('code'),
        state: params.get('state'),
        error: params.get('error'),
      }
    }

    export async function exchangeCodeForToken(code: string, endpoint: string): Promise<string> {
      const res = await fetch(endpoint, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ code }),
      })
      if (!res.ok) throw new Error(`exchange_failed_${res.status}`)
      const data = (await res.json()) as { access_token?: string; error?: string }
      if (!data.access_token) throw new Error(data.error || 'exchange_failed')
      return data.access_token
    }

    export async function fetchGitHubUser(token: string): Promise<GitHubUser> {
      const res = await fetch('https://api.github.com/user', {
        headers: {
          Authorization: `Bearer ${token}`,
          Accept: 'application/vnd.github+json',
        },
      })
      if (!res.ok) throw new Error(`user_fetch_failed_${res.status}`)
      const data = (await res.json()) as GitHubUser
      return { login: data.login, avatar_url: data.avatar_url }
    }
    ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| app -> github.com/login/oauth/authorize | Redirect carries the CSRF `state`; the returned callback must be verified against it |
| app -> VITE_AUTH_ENDPOINT | Client sends `code`; endpoint host is config-driven and untrusted infrastructure-wise |
| app -> api.github.com/user | Bearer token sent to fetch identity |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-11-06 | Spoofing (CSRF) | OAuth callback | mitigate | `generateState()` produces a 128-bit random nonce embedded in the authorize URL; the hook (11-03) verifies it on callback. This lib supplies the primitive; the consumer enforces it. |
| T-11-07 | Tampering | exchange request | mitigate | `code` sent as a JSON body field to a fixed, config-provided endpoint over HTTPS; not concatenated into a URL. |
| T-11-08 | Information Disclosure | access token in transit | accept | Token travels over HTTPS to api.github.com; at-rest storage risk (localStorage/XSS) accepted per CONTEXT.md D-05/D-10 and handled in 11-03. |
</threat_model>
