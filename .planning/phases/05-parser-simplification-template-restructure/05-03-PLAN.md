**Step A: Replace Preview.tsx**

New Preview.tsx renders only a container div with dangerouslySetInnerHTML (per D-03, D-04):

```tsx
import { TEMPLATE_STYLES, type TemplateName } from '../lib/templateStyles'

interface PreviewProps {
  htmlContent: string
  template: TemplateName
}

export default function Preview({ htmlContent, template }: PreviewProps) {
  const styles = TEMPLATE_STYLES[template]

  if (!htmlContent.trim()) {
    return (
      <div className="flex items-center justify-center h-full text-gray-400 text-sm">
        Start typing markdown to see your resume preview
      </div>
    )
  }

  return (
    <div
      className={styles.container}
      dangerouslySetInnerHTML={{ __html: htmlContent }}
    />
  )
}
```

Note: Only `styles.container` is applied in Phase 5 (per D-03, D-04). Element-level class injection is Phase 6.
