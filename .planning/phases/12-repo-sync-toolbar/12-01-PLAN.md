export async function fetchGitHubUser(token: string): Promise<GitHubUser> {
  const res = await fetch('https://api.github.com/user', {

}

Add module-private base64 helpers that handle UTF-8 correctly (btoa/atob alone corrupt non-ASCII — DO NOT use them directly on the string):

```typescript
function decodeBase64(b64: string): string {
  const clean = b64.replace(/\n/g, '')
  const bytes = Uint8Array.from(atob(clean), (c) => c.charCodeAt(0))
  return new TextDecoder().decode(bytes)
}
function encodeBase64(text: string): string {
  const bytes = new TextEncoder().encode(text)
  let binary = ''
  for (let i = 0; i < bytes.length; i++) binary += String.fromCharCode(bytes[i])
  return btoa(binary)
}
function encodePath(path: string): string {
  return path.split('/').map(encodeURIComponent).join('/')
}
```

|----------|-------------|
| app → api.github.com | The user's OAuth token (repo scope) is attached to every request; responses become editor content |
| user-selected path/branch → fetch URL | Repo, branch, and file path strings are interpolated into API URLs |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-12-01 | Information Disclosure | token in fetch headers | mitigate | Token only ever sent in `Authorization` header to hardcoded `https://api.github.com` URLs; no `console.log` of token (grep-verified); no fetch to any other host |
| T-12-02 | Tampering | path/branch interpolated into URL | mitigate | `encodePath()` URL-encodes each path segment; branch encoded via `encodeURIComponent` — prevents path traversal / query injection into the API URL |
| T-12-03 | Elevation of Privilege | repo scope token misuse | accept | Token scope is fixed at Phase 11 (`repo`); this module performs only read + single-file PUT the user explicitly initiates; no broader operations exposed |
</threat_model>
