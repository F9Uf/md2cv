    Create src/components/FileTree.tsx — a controlled recursive tree. Props:
    ```typescript
    import type { TreeNode } from '../lib/fileTree'
    interface FileTreeProps {
      nodes: TreeNode[]
      activePath: string | null      // repoConfig.filePath
      isDirty: boolean
      expandedPaths: Set<string>
      onToggleFolder: (path: string) => void
      onOpenFile: (path: string) => void  // called only for .md file rows
      depth?: number                 // default 0, used for indentation
    }
    ```
    Rendering rules (from UI-SPEC section 3):
    - Wrap the top-level list in `<ul role="tree" aria-label="File tree">`; nested folder children in `<ul role="group">`.
    - Each node is a `<li role="treeitem">`.
      - Folders: `aria-expanded={expandedPaths.has(node.path)}`. Render a full-width `<button>` row;
        onClick calls `onToggleFolder(node.path)`. Show a 12×12 chevron SVG with
        `className={\`transition-transform duration-150 ${expanded ? 'rotate-90' : ''}\`}` then a 14×14
        folder icon, then the name. When expanded, recursively render `<FileTree ... nodes={node.children} depth={depth+1} />` inside a nested `<ul role="group">`.
      - Files:
        - `.md` file (`node.isMarkdown === true`): render a full-width `<button>` row,
          `onClick={() => onOpenFile(node.path)}`, classes `text-sm text-gray-200 hover:bg-gray-700 cursor-pointer`,
          `aria-selected={node.path === activePath}`. A file has NO chevron — insert a 12px-wide spacer
          so the file icon aligns with folder icons. File icon 14×14.
        - non-`.md` file (`node.isMarkdown === false`): render a NON-interactive `<div>` row with
          `aria-disabled="true"`, classes `text-sm text-gray-500 cursor-not-allowed` and NO hover class,
          NO onClick.
    - Indentation: apply `style={{ paddingLeft: depth * 12 + 8 }}` to each row element.
    - Row height: base classes include `flex items-center h-7 w-full text-left`; add responsive height
      via `max-[767px]:h-11` so mobile rows meet the 44px touch target (UI-SPEC Spacing exceptions).
    - Active row: when `node.path === activePath`, add `bg-gray-700` to the row and render the name as
      `text-white font-semibold`. On the active `.md` row ONLY, when `isDirty`, append the dirty dot:
      `<span className="w-1.5 h-1.5 rounded-full bg-amber-400 ml-auto mr-2" aria-hidden="true" />`.
    - Name text uses `truncate` to clip long names within the 240px sidebar.
    - Chevron / folder / file icons are hand-authored inline SVGs (viewBox="0 0 16 16" scaled to 12 or
      14 px), `fill="currentColor"`, `aria-hidden="true"`, `text-gray-400`. Use a right-pointing
      triangle for the chevron (rotates to point down when expanded), a simple folder glyph, and a
      simple document glyph for files.
    - The component is recursive: it renders itself for each expanded folder's children with `depth+1`.
  </action>

    ```tsx
    import Dialog from './Dialog'

    interface DirtySwitchDialogProps {
      open: boolean
      currentFilename: string
      committing: boolean
      onCommit: () => void     // opens the Phase 12 CommitDialog flow
      onDiscard: () => void
      onCancel: () => void
    }

    export default function DirtySwitchDialog({
      open, currentFilename, committing, onCommit, onDiscard, onCancel,
    }: DirtySwitchDialogProps) {
      return (
        <Dialog open={open} title="Unsaved Changes" onClose={onCancel}>
          <p className="text-sm text-gray-300 mb-6">
            <strong>{currentFilename}</strong> has uncommitted changes. What would you like to do?
          </p>
          <div className="flex flex-col gap-2">
            <button
              onClick={onCommit}
              disabled={committing}
              className="h-9 w-full rounded bg-blue-600 text-white text-sm border border-blue-500 hover:bg-blue-500 transition-colors disabled:opacity-60 disabled:cursor-not-allowed"
            >
              Commit changes…
            </button>
            <button
              onClick={onDiscard}
              className="h-9 w-full rounded text-red-400 text-sm hover:bg-gray-800 transition-colors"
            >
              Discard my edits
            </button>
            <button
              onClick={onCancel}
              className="h-9 w-full rounded bg-gray-700 text-white text-sm border border-gray-600 hover:bg-gray-600 transition-colors"
            >
              Cancel
            </button>
          </div>
        </Dialog>
      )
    }
    ```
